export const registerUser = async (req, res, next) => {
    try {
        const { name, phone_number, role } = req.body;

        if (!name || !phone_number || !role) {
            throw new ValidationError("All fields are required");
        }

        validateRegistrationData(phone_number);

        // OPTIMIZATION: Start checking constraints immediately (Parallel)
        // 1. Check if user exists in DB
        const userCheckPromise = (role === "CUSTOMER") 
            ? prisma.user_customer.findUnique({ where: { phone_number } })
            : prisma.electrician_customer.findUnique({ where: { phone_number } });

        // 2. Check if user is blocked in Redis (Spam/Lock)
        // We call this here so it runs WHILE the DB is searching
        const restrictionCheckPromise = checkOtpRestrication(phone_number);

        // Await both results
        const [existingUser, _] = await Promise.all([
            userCheckPromise,
            restrictionCheckPromise
        ]);

        if (existingUser) {
            throw new ValidationError("User already exists with this phone number");
        }

        // 3. Track request count
        await trackOtpRequests(phone_number);

        // 4. Save Temp User Info & Send OTP
        // run these in parallel too
        await Promise.all([
             redis.set(`info:${phone_number}`, JSON.stringify({ name, role }), "EX", 300),
             sendotpcode(phone_number)
        ]);
        
        // Response sent only after everything is confirmed
        res.status(200).json({
            success: true,
            message: "OTP sent successfully"
        });

    } catch (error) {
        next(error);
    }
};
